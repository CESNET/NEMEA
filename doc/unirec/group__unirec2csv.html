Detailed Description

Functions to convert UniRec template and data into CSV-like representation

urcsv_t *csv = urcsv_init(tmplt, ',');
char *str = urcsv_header(csv);
fprintf(stderr, "%s\n", str);
free(str);
str = urcsv_record(csv, rec);
fprintf(stderr, "%s\n", str);
free(str);
urcsv_free(&csv);
