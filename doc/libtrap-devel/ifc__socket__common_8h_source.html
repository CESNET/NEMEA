133 static inline void insert_into_buffer(buffer_t *buffer, const void *data, uint16_t size)
134 {
135  uint16_t *msize = (uint16_t *)(buffer->data + buffer->wr_index);
136  (*msize) = htons(size);

138  buffer->wr_index += (size + sizeof(size));
